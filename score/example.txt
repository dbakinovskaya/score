Пример заполнения базы данных моделью H2H:


from myapp.models import H2H, Teams, Fixtures, Leagues
import datetime

team1 = Teams.objects.get(team_name='Arsenal')
team2 = Teams.objects.get(team_name='Manchester City')
fixture = Fixtures.objects.filter(home_team_id=team1, away_team_id=team2)
league = Leagues.objects.get(league_name='Premier League')
fixture = Fixtures.objects.get(id=1)
h2h = H2H(team1_id=team1, team2_id=team2, date=datetime.date.today(), fixture_id=fixture, league_id=league)
h2h.save()
