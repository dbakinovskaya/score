Пример заполнения базы данных моделью Fixtures:


from myapp.models import Fixtures, Seasons, Leagues, Teams
import datetime

season = Seasons.objects.get(name='2021/2022')
league = Leagues.objects.get(name='Premier League')
home_team = Teams.objects.get(name='Manchester United')
away_team = Teams.objects.get(name='Liverpool')

fixture = Fixtures(seasons_id=season, league_id=league, home_team_id=home_team, away_team_id=away_team, date=datetime.datetime.now())
fixture.save()


Пример заполнения базы данных моделью H2H:


from myapp.models import H2H, Teams, Fixtures, Leagues
import datetime

team1 = Teams.objects.get(name='Manchester United')
team2 = Teams.objects.get(name='Liverpool')
fixture = Fixtures.objects.get(home_team_id=team1, away_team_id=team2)
league = Leagues.objects.get(name='Premier League')

h2h = H2H(team1_id=team1, team2_id=team2, date=datetime.date.today(), fixture_id=fixture, league_id=league)
h2h.save()

from myapp.models import Live, Fixtures

fixture = Fixtures.objects.get(id=1)

live = Live(fixture_id=fixture, home_score=1, away_score=0)
